#include <QtCore>
#include "xlsxdocument.h"

int main()
{
    QXlsx::Document xlsx;

    xlsx.write("A1", "Hello Qt!");
    xlsx.write("A2", 12345);
    xlsx.write("A3", "=44+33");
    xlsx.write("A4", true);
    xlsx.write("A5", "http://qt-project.org");
    xlsx.write("A6", QDate(2013, 12, 27));
    xlsx.write("A7", QTime(6, 30));

    xlsx.save();

    return 0;
}
